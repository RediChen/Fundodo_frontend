Table users{
    id 	int(6)
    name	        varchar(50)
    nickname	    varchar(50)
    account	        varchar(50)
    password_hash	varchar(60)
    gender		    tinyint(1)
    user_level		tinyint(1)
    dob             date
    tel 	        varchar(20)
    email           varchar(50)
    avatar_file     varchar(150)
    address         varchar(100)
    adr_city        varchar(100)
    adr_district    int(3)
    created_at  	datetime	
    deleted_at      datetime
}



Table user_level {
  id 	tinyint(2) [primary key]
  name varchar(10)
}
