//購物車、訂單、優惠券

Table carts {
  id integer [primary key]
  user_id integer//users 的 id
  prod_item_id integer//prod_price_stock 的 id
  crs_id integer//course 的 id
  qty integer
  created_at datetime
}

Table orders {
  id integer [primary key]
  user_id integer//users 的 id
  prod_item_id integer//prod_price_stock 的 id
  crs_id integer//course 的 id
  qty integer
  created_at datetime
}
